# Migrate from 3.14.1 to 3.15.2

## Breaking changes

### `bytebase_user` changes

- Remove the `type` field.
- Separate the `SERVICE_ACCOUNT` to new resource `bytebase_service_account`.
- The service account can be managed at the workspace-level or project-level.
- The service account MUST use `serviceAccount:` prefix in the IAM binding.

Before:

```terraform
# service account
resource "bytebase_user" "ci_bot" {
  title = "CI Bot"
  email = "ci@service.example.com"
  type  = "SERVICE_ACCOUNT"
}

# end user
resource "bytebase_user" "workspace_dba" {
  email = "dba@example.com"
  title = "Database Administrator"
  type  = "USER"
}

# IAM binding for service account
resource "bytebase_iam_policy" "project_iam" {
  parent = "projects/sample"

  iam_policy {
    binding {
      role = "roles/projectViewer"
      members = [
        "user:ci@service.example.com"
      ]
    }
  }
}
```

After:

```terraform
# service account
resource "bytebase_service_account" "ci_bot" {
  parent             = "workspaces/-" # parent can be "workspaces/-" or "projects/{project}"
  service_account_id = "terraform"    # The service account email will be "{id}@service.bytebase.com"
  title              = "CI Bot"
}

# end user
resource "bytebase_user" "workspace_dba" {
  email = "dba@example.com"
  title = "Database Administrator"
}

# IAM binding for service account
resource "bytebase_iam_policy" "project_iam" {
  parent = "projects/sample"

  iam_policy {
    binding {
      role = "roles/projectViewer"
      members = [
        format("serviceAccount:%s", bytebase_service_account.ci_bot.email),
      ]
    }
  }
}
```

### Deprecate the `data_source_query_policy`

- Remove the entier `data_source_query_policy` entity.
- The `restriction` field is simplified as the `allow_admin_data_source` in the `query_data_policy`.
- The `disallow_ddl`/`disallow_dml` are removed. Use the IAM binding condition with `environment_ids` to control DDL/DML access per environment.

Before:

```terraform
resource "bytebase_policy" "data_source_query_policy" {
  parent     = "environments/test"
  type       = "DATA_SOURCE_QUERY"

  data_source_query_policy {
    restriction  = "FALLBACK"
    # DDL/DML is allowed for the test environment.
    disallow_ddl = false
    disallow_dml = false
  }
}
```

After:

```terraform
resource "bytebase_policy" "query_data_policy" {
  parent = "workspaces/-"
  type   = "DATA_QUERY"
  query_data_policy {
    maximum_result_rows     = 100
    disable_export          = true
    disable_copy_data       = true
    # not allow using the admin data source if exists the read-only data source.
    allow_admin_data_source = false
  }
}

# Use IAM binding condition with environments to control DDL/DML access
resource "bytebase_iam_policy" "project_iam" {
  parent = "projects/sample"

  iam_policy {
    binding {
      # role with bb.sql.ddl or bb.sql.dml permission
      role = "roles/sqlEditorUser"
      members = [
        "user:sample@bytebase.com"
      ]
      condition {
        # allow DDL/DML in the following environments
        environment_ids = [
          "test"
        ]
      }
    }
  }
}
```

### Update the `query_data_policy`

Related changes: update the `workspace_profile` setting

- The `query_data_policy` can only be configured on the `workspaces/-` or `projects/{id}`. No more environment-level policy.
- Move the `maximum_result_size` to the `workspace_profile` setting as `sql_result_size` field.
- Move the `timeout_in_seconds` to the `workspace_profile` setting as `query_timeout_in_seconds` field.
- Add the `allow_admin_data_source` field.
- Add the `disable_copy_data` field.

Before:

```terraform
# workspace-level policy
resource "bytebase_policy" "query_data_policy" {
  parent = "workspaces/-"
  type   = "DATA_QUERY"
  query_data_policy {
    maximum_result_size = 200 * 1024 * 1024 # 200MB
    maximum_result_rows = 100
    disable_export      = false
    timeout_in_seconds  = 60 # 60 seconds
  }
}

# environment-level policy
resource "bytebase_policy" "env_query_data_policy" {
  parent     = "environments/prod"
  type       = "DATA_QUERY"

  query_data_policy {
    disable_copy_data = true
  }
}
```

After:

```terraform
resource "bytebase_policy" "query_data_policy" {
  parent = "workspaces/-"
  type   = "DATA_QUERY"
  query_data_policy {
    maximum_result_rows     = 100
    disable_export          = false
    disable_copy_data       = true
    allow_admin_data_source = false
  }
}

resource "bytebase_setting" "workspace_profile" {
  name = "settings/WORKSPACE_PROFILE"

  workspace_profile {
    sql_result_size          = 200 * 1024 * 1024 # 200MB
    query_timeout_in_seconds = 60                # 60 seconds
  }
}
```

### Update `bytebase_project`

- Remove the `auto_enable_backup` field.
- Remove the `skip_backup_errors` field.

## Features

### Support `bytebase_service_account`

Support manage the service account at the workspace-level or project-level.

- Resource: `bytebase_service_account`
- Data sources: `bytebase_service_account`, `bytebase_service_account_list`

Examples:

```terraform
resource "bytebase_service_account" "ci_bot" {
  parent             = "workspaces/-"
  service_account_id = "terraform"
  title              = "CI Bot"
}

# Read a service account
data "bytebase_service_account" "ci_bot" {
  name = bytebase_service_account.ci_bot.name
}

# List all service accounts
data "bytebase_service_account_list" "all" {
  parent = "workspaces/-"
}

# IAM binding for service account
resource "bytebase_iam_policy" "workspace_iam" {
  parent = "workspaces/-"

  iam_policy {
    binding {
      role = "roles/workspaceAdmin"
      members = [
        format("serviceAccount:%s", bytebase_service_account.ci_bot.email),
      ]
    }
  }
}
```

### Support `bytebase_workload_identity`

Support manage the workload identity at the workspace-level or project-level.

- Resource: `bytebase_workload_identity`
- Data sources: `bytebase_workload_identity`, `bytebase_workload_identity_list`
- The workload identity MUST use `workloadIdentity:` prefix in the IAM binding.

Examples:

```terraform
# Create or update the workload identity for GitHub Actions CI/CD.
resource "bytebase_workload_identity" "github_ci" {
  parent               = "workspaces/-"
  workload_identity_id = "github-ci"
  title                = "GitHub CI"

  workload_identity_config {
    provider_type   = "GITHUB"
    subject_pattern = "repo:bytebase/sample:ref:refs/heads/main"
  }
}

# IAM binding for workload identity
resource "bytebase_iam_policy" "workspace_iam" {
  parent = "workspaces/-"

  iam_policy {
    binding {
      role = "roles/workspaceAdmin"
      members = [
        format("workloadIdentity:%s", bytebase_workload_identity.github_ci.email),
      ]
    }
  }
}
```
